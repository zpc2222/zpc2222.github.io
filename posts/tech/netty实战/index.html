    private static List<String> getIpAddress() {

        try {

                String name = intf.getName();

                    for (Enumeration<InetAddress> enumIpAddr = intf.getInetAddresses(); enumIpAddr.hasMoreElements(); ) {

                    }  
                }  
            }

            String ip = "127.0.0.1";  
            list.add(ip);  
        }  
        return list;
